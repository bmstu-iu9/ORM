CreateTableStmt: CREATE TABLE (IF NOT EXISTS)?
                 qualified_name '(' TableElementList? ')' Inherit?

Inherit:     INHERITS '(' qualified_name (',' qualified_name)* ')'

qualified_name: ColId ('.'ColId)*
ColId: IDENT | unreserved_keyword | col_name_keyword

TableElementList: TableElement (',' TableElement)*
TableElement:     columnDef | TableConstraint
columnDef: ColId Typename ColConstraint*

Typename:  SimpleTypename
SimpleTypename: Numeric
                | CharacterType
                | ConstDatetime

Numeric: INT
         | INTEGER
         | SMALLINT
         | BIGINT
         | REAL
         | FLOAT  ( '('intConst')' )? //1 <= intConst <= 53
         | DOUBLE PRECISION
         | DECIMAL
         | NUMERIC
         | BOOLEAN

CharacterType: character ( '(' intConst ')' )?

character: CHARACTER
           | CHAR
           | VARCHAR

ConstDatetime: TIMESTAMP ( '(' intConst ')' )? //0 <= intConst < 6
               | TIME ( '(' intConst ')' )?
               | DATE

TableConstraint: CONSTRAINT ColId ConstraintElem
                 | ConstraintElem

ConstraintElem:  UNIQUE        '(' ColId (',' ColId)* ')'
                 | PRIMARY KEY '(' ColId (',' ColId)* ')'
                 | FOREIGN KEY '(' ColId (',' ColId)* ')' REFERENCES qualified_name
                   opt_column_list key_actions

ColConstraint:  CONSTRAINT ColId ColConstraintElem
                | ColConstraintElem

ColConstraintElem: NOT NULL
                   | NULL
                   | UNIQUE
                   | PRIMARY KEY
                   | CHECK '(' b_expr ')'     //HERE NEED TO CHECK APPLICATION OF OPs
                   | DEFAULT expr_no_vars     //ARITHMETIC ONLY EXPR OR VALUE TILL
                   | REFERENCES qualified_name opt_column_list key_actions

opt_column_list:   ( '(' ColId (',' ColId)* ')' )?

key_actions: ON UPDATE key_action
             | ON DELETE key_action
             | ON UPDATE key_action ON DELETE key_action
             | ON DELETE key_action ON UPDATE key_action
             | /*EMPTY*/

key_action:  NO ACTION
             | RESTRICT
             | CASCADE
             | SET NULL
             | SET DEFAULT

expr_no_vars: an_expr | NumericValue | TRUE | FALSE | CharacterValue | DateValue

an_expr: an_term ( {'+' | '-'} an_term )*
an_term: an_factor ( {'*' | '/'} an_factor )*
an_factor: NumericValue
          | '-' an_factor
          | '(' an_expr ')'

a_expr: a_term ( {'+' | '-'} a_term )*
a_term: a_factor ( {'*' | '/'} a_factor )*
a_factor: ColId
          | NumericValue
          | '-' a_factor
          | '(' a_expr ')'

b_stmt: a_expr   '<'  a_expr
        | a_expr '<=' a_expr
        | a_expr '>'  a_expr
        | a_expr '>=' a_expr
        | a_expr '='  a_expr
        | a_expr '!=' a_expr
        | a_expr IS NULL
        | a_expr IS NOT NULL
        | a_expr IS TRUE
        | a_expr IS NOT TRUE
        | a_expr IS FALSE
        | a_expr IS NOT FALSE
        | ColId LIKE StringValue               // ONLY FOR STRING TYPE OF ColId
        | StringValue LIKE ColId               // ONLY FOR STRING TYPE OF ColId
        | ColId NOT LIKE StringValue           // ONLY FOR STRING TYPE OF ColId
        | StringValue NOT LIKE ColId           // ONLY FOR STRING TYPE OF ColId
        | a_expr BETWEEN b_expr AND a_expr     // ARITHMETIC OR DATE ONLY TILL
        | a_expr NOT BETWEEN a_expr AND a_expr // ARITHMETIC OR DATE ONLY TILL

b_expr: b_term (OR b_term)*
b_term: b_factor (AND b_factor)*
b_factor: b_const | NOT b_factor | '(' b_expr ')'
b_const: ColId | TRUE | FALSE | NULL | b_stmt
